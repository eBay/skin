Required JavaScript

There are 2 ways to implement floating labels, via. CSS or JS. The CSS solution had two constraints, label had to come after input tags and required hacks to make it work in IE/Edge. So Skin decided to use the JavaScript option.

First the --floated modifier must be applied to label tag. This was done to support the case where JavaScript is disabled. In this case, the label would always be floated to allow users to interact with the input. In case JavaScript is enabled, on page load, the above modifier would be kept if the input has a default value or removed if the input had none. The --floated modifier is then toggled using event listeners for focus & blur.

An example implementation is shown below.

{% highlight js %} window.addEventListener('load', function() { toggleFloatingLabels(); }); function toggleFloatingLabels() { const inputEls = document.querySelectorAll('.floated-label .textbox__control'); inputEls.forEach(function(input) { const siblingLabel = input.parentNode.parentNode.querySelector('.floated-label__label'); if (!input.value) { siblingLabel.classList.remove('floated-label__label--floated'); } }); } const inputWrapper = document.querySelector('#floated-label'); inputWrapper.addEventListener('focus', function (event) { const inputEl = event.target; if (!inputEl.classList.contains('textbox__control')) { event.stopPropagation(); } const siblingLabel = inputEl.parentNode.parentNode.querySelector('.floated-label__label'); siblingLabel.classList.add('floated-label__label--floated'); }, true); inputWrapper.addEventListener('blur', function (event) { const inputEl = event.target; if (!inputEl.classList.contains('textbox__control')) { event.stopPropagation(); } const siblingLabel = inputEl.parentNode.parentNode.querySelector('.floated-label__label'); if (inputEl.value !== '') { siblingLabel.classList.add('floated-label__label--floated'); } else { siblingLabel.classList.remove('floated-label__label--floated'); } }, true); {% endhighlight %}
